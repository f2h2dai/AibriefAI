from __future__ import annotations

import argparse
import json
import mimetypes
import os
from datetime import datetime, timezone
from http import HTTPStatus
from http.server import SimpleHTTPRequestHandler, ThreadingHTTPServer
from pathlib import Path
from typing import Any

DEFAULT_HEALTH_MAX_AGE_SECONDS = 600


def _parse_dt(value: str) -> datetime | None:
    if not value:
        return None
    try:
        return datetime.fromisoformat(value.replace("Z", "+00:00"))
    except ValueError:
        return None


def _read_json(path: Path) -> dict[str, Any]:
    if not path.exists():
        return {}
    return json.loads(path.read_text(encoding="utf-8"))


def build_health(report_path: Path, max_age_seconds: int) -> tuple[int, dict[str, Any]]:
    now = datetime.now(timezone.utc)
    report = _read_json(report_path)
    started_at = _parse_dt(str(report.get("started_at", "")))
    if started_at is None:
        return HTTPStatus.SERVICE_UNAVAILABLE, {
            "status": "fail",
            "reason": "missing_or_invalid_pipeline_timestamp",
            "checked_at": now.isoformat(),
        }
    age_seconds = max(0, int((now - started_at).total_seconds()))
    ok = age_seconds < max_age_seconds and not report.get("errors")
    return (HTTPStatus.OK if ok else HTTPStatus.SERVICE_UNAVAILABLE), {
        "status": "ok" if ok else "fail",
        "checked_at": now.isoformat(),
        "run_id": report.get("run_id", ""),
        "topic": report.get("topic", ""),
        "pipeline_started_at": started_at.isoformat(),
        "pipeline_age_seconds": age_seconds,
        "max_age_seconds": max_age_seconds,
        "metrics": report.get("metrics", {}),
        "error_count": len(report.get("errors", [])),
    }


class AibriefHandler(SimpleHTTPRequestHandler):
    server_version = "AibriefHTTP/1.0"

    def end_headers(self) -> None:
        self.send_header("X-Content-Type-Options", "nosniff")
        self.send_header("X-Frame-Options", "DENY")
        self.send_header("Referrer-Policy", "no-referrer")
        self.send_header("Permissions-Policy", "geolocation=(), microphone=(), camera=()")
        super().end_headers()

    def do_GET(self) -> None:  # noqa: N802
        if self.path.split("?", 1)[0].rstrip("/") == "/health":
            status, body = build_health(self.server.report_path, self.server.health_max_age_seconds)  # type: ignore[attr-defined]
            payload = json.dumps(body, ensure_ascii=False, indent=2).encode("utf-8")
            self.send_response(status)
            self.send_header("Content-Type", "application/json; charset=utf-8")
            self.send_header("Cache-Control", "no-store")
            self.send_header("Content-Length", str(len(payload)))
            self.end_headers()
            self.wfile.write(payload)
            return
        super().do_GET()


def main(argv: list[str] | None = None) -> int:
    parser = argparse.ArgumentParser(description="Serve Aibrief dashboard with dynamic /health")
    parser.add_argument("--host", default=os.getenv("HOST", "0.0.0.0"))
    parser.add_argument("--port", type=int, default=int(os.getenv("PORT", "8000")))
    parser.add_argument("--web-root", default=os.getenv("AIBRIEF_WEB_ROOT", "web"))
    parser.add_argument("--report-path", default=os.getenv("AIBRIEF_REPORT_PATH", "data/latest_report.json"))
    parser.add_argument("--health-max-age-seconds", type=int, default=int(os.getenv("AIBRIEF_HEALTH_MAX_AGE_SECONDS", str(DEFAULT_HEALTH_MAX_AGE_SECONDS))))
    args = parser.parse_args(argv)

    web_root = Path(args.web_root).resolve()
    report_path = Path(args.report_path).resolve()
    if not web_root.exists():
        raise SystemExit(f"missing web root: {web_root}")
    os.chdir(web_root)
    mimetypes.add_type("application/json", ".json")
    server = ThreadingHTTPServer((args.host, args.port), AibriefHandler)
    server.report_path = report_path  # type: ignore[attr-defined]
    server.health_max_age_seconds = int(args.health_max_age_seconds)  # type: ignore[attr-defined]
    print(f"Aibrief dashboard serving http://{args.host}:{args.port} from {web_root}")
    server.serve_forever()
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
